cin>>x;
int a[x];
for(int i=0;i<x;i++)
{ cin>>a[i]; } // دخلت الارقام اللى هبحث فيها
int y; // الرقم اللى بنبحث عنه
cin>>y;
for(int i=0;i<=x/2;i++)
{
    if(y==a[i])
       {cout<<"YES\n";
           return ;
                  }
    if(y==a[x/2 + i])
     {cout<<"YES\n";
         return ;
                }
}
